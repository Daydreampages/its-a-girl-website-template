  <!-- Countdown Timer Script -->
  <script>
    var countDownDate = new Date("Aug 18, 2025 00:00:00").getTime();
    var x = setInterval(function () {
      var now = new Date().getTime();
      var distance = countDownDate - now;
      var days = Math.floor(distance / (1000 * 60 * 60 * 24));
      var hours = Math.floor((distance % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60));
      var minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));
      var seconds = Math.floor((distance % (1000 * 60)) / 1000);
      document.getElementById("countdown").innerHTML =
        days + "d " + hours + "h " + minutes + "m " + seconds + "s ";
      if (distance < 0) {
        clearInterval(x);
        document.getElementById("countdown").innerHTML = "Today is the day!";
      }
    }, 1000);
  </script>

  <!-- Background Slideshow Script -->
  <script>
    window.addEventListener('load', function () {
      const backgrounds = [
        'assets/girl-bg-1.jpg',
        '#FFDDEE',
        'assets/girl-bg-2.jpg',
        '#FFF0F5',
        'assets/girl-bg-3.jpg',
        '#FFE4EC'
      ];

      let current = 0;
      const backgroundDiv = document.getElementById('background');

      function changeBackground() {
        backgroundDiv.style.transition = "opacity 3s ease-in-out";
        backgroundDiv.style.opacity = 0;
        setTimeout(() => {
          const bg = backgrounds[current];
          if (bg.startsWith('#')) {
            backgroundDiv.style.backgroundImage = 'none';
            backgroundDiv.style.backgroundColor = bg;
            backgroundDiv.style.animation = "pulse 5s ease-in-out infinite";
          } else {
            backgroundDiv.style.backgroundImage = `url('${bg}')`;
            backgroundDiv.style.backgroundColor = 'transparent';
            backgroundDiv.style.animation = "none";
          }
          backgroundDiv.style.opacity = 1;
          current = (current + 1) % backgrounds.length;
        }, 2000);
      }

      changeBackground();
      setInterval(changeBackground, 10000);

      const loader = document.getElementById('loader');
      loader.style.opacity = '0';
      setTimeout(function () {
        loader.style.display = 'none';
      }, 500);
    });
  </script>
